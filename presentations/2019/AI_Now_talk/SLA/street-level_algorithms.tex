\begin{frame}\frametitle{street-level algorithms}

\visible<1->{Street-level algorithms are algorithmic systems that directly interact with and make decisions about people in a sociotechnical system.}

\visible<2->{
  They classify and recommend:
  \begin{itemize}
    \item the order and visibility of posts on your news feed
    \visible<3>{{\small(\cite{Rader:2015:UUB:2702123.2702174,bucher2017algorithmic,bozdag2015breaking,eslami2016first,Eslami:2015:IAA:2702123.2702556})}}
    \item eligibility to contribute to peer production sites
    \visible<3>{{\small(\cite{wikipediansBornNotMade,geiger2018lives})}}
    \item where and when we get work
    \visible<3>{{\small(\cite{uberAlgorithm})}}
  \end{itemize}
}
